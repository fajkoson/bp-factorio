cmake_minimum_required(VERSION 3.16)
project(bpbook VERSION 1.0 LANGUAGES CXX)

# Set C++ standard and define necessary options
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# Precompiled headers option
option(USE_PRECOMPILED_HEADERS "Use precompiled headers for faster compilation" ON)

# External dependencies (e.g., nlohmann::json)
include(FetchContent)
FetchContent_Declare(
    json
    GIT_REPOSITORY https://github.com/nlohmann/json.git
    GIT_TAG v3.10.5
)
FetchContent_MakeAvailable(json)

# Include directories
include_directories(${PROJECT_SOURCE_DIR}/include)

# Add subdirectories
add_subdirectory(src)
add_subdirectory(precompiled)

# Optionally add tests
if (BPBOOK_ENABLE_TESTING)
    enable_testing()
    add_subdirectory(tests)
endif()

# Install targets
install(TARGETS bpbook DESTINATION bin)
install(DIRECTORY include/bpbook DESTINATION include)
